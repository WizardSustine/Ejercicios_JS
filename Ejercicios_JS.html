  10. Escribe un programa que pida una frase y escriba las vocales que aparecen.

    function ej10(){
    let frase = prompt("Escriba una frase");
    frase = frase.toLowerCase();
    let arrayvocales = [];
    for(ii = 0; ii <= frase.length; ii++){
    switch(frase.charAt(ii)){
    case "a":
    arrayvocales.push("a");
    break;
    case "e":
    arrayvocales.push("e");
    break;
    case "i":
    arrayvocales.push("i");
    break;
    case "o":
    arrayvocales.push("o");
    break;
    case "u":
    arrayvocales.push("u");
    break;
    default:
    break;
    }
    }
    alert("Las siguiente letras son vocales: " + removeDuplicate(arrayvocales));
    }
    function removeDuplicate(arrayvocales){
    return [...new Set(arrayvocales)];
    }
